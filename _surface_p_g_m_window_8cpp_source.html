379 void SurfacePGMWindow::load_surface_pgm(){

381 
382  QString path = QFileDialog::getOpenFileName(this, tr("Open Surface PGM File"),

393  body_fixed_acc_magnitudes,

396  arma::vec::fixed<3> omega;
397 
398  try{

404  slopes,

422  return;

431

437  this -> parent -> get_renderer() -> RemoveActor2D(wrapper -> get_colorbar_actor());
438 
439  this -> parent -> qvtkWidget -> GetRenderWindow() -> Render();
440  this -> parent -> log_console -> appendPlainText(QString::fromStdString("\n- Done loading surface PGM from file " + path.toStdString()));
441 
442  }

444  QMessageBox::warning(this, "Load Surface PGM", "An error occured loading the Surface PGM file.");
