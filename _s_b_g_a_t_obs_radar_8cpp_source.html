117 void SBGATObsRadar::reverse_ray_trace(SBGATRadarObsSequence & measurements_sequence,
118  const std::vector<std::vector<int> > & facets_in_view,
119  const arma::vec & radar_dir,
120  const int N,
121  const bool penalize_indicence,
122  const std::vector<arma::mat> & BN_dcms_vec,
123  const std::vector<arma::vec> & positions_vec,
124  const std::vector<arma::vec> & velocities_vec,
125  const std::vector<arma::vec> & omegas_vec){
126 
127  std::vector<std::array<double, 3> > measurements;
128 
129  // Ray-tracing tolerance
130  double tol = this -> polydata_vec[0] -> GetLength()/1E6;
131 
132  // The radar is positionned with respect to the primary
133  arma::vec radar_pos = this -> center_of_mass_vec[0] + this -> polydata_vec[0] -> GetLength() * 1E6 * radar_dir;
134 
135  for (int body_index = 0; body_index < this -> number_of_bodies; ++body_index){
136 
137  vtkPolyData * input = this -> polydata_vec[body_index];
138 
139  vtkSmartPointer<vtkIdList> ptIds = vtkSmartPointer<vtkIdList>::New();
140  ptIds -> Allocate(VTK_CELL_SIZE);
141 
142  arma::vec target_to_radar_dir_body_frame = BN_dcms_vec[body_index] * radar_dir;
143 
144 
145  // The kept facets are then sampled and reverse ray-traced
146  for (unsigned int facet_index = 0; facet_index != facets_in_view[body_index].size(); ++facet_index){
147 
148  double p0[3];
149  double p1[3];
150  double p2[3];
151 
152  input -> GetCellPoints(facets_in_view[body_index][facet_index],ptIds);
153  input -> GetPoint(ptIds->GetId(0), p0);
154  input -> GetPoint(ptIds->GetId(1), p1);
155  input -> GetPoint(ptIds->GetId(2), p2);
156 
157  arma::vec P0 = {p0[0],p0[1],p0[2]};
158  arma::vec P1 = {p1[0],p1[1],p1[2]};
159  arma::vec P2 = {p2[0],p2[1],p2[2]};
160 
161  arma::vec n = arma::cross(P1 - P0, P2 - P0);
162 
163  // The number of points sampled from this facet is determined based on
164  // the relative size of this facet compared to the largest one in all the considered shapes
165  int N_samples = int( N * arma::norm(n /2) / this -> min_area);
166 
167  // only need unit normal vector from here
168  n = arma::normalise(arma::cross(P1 - P0, P2 - P0));
169 
170  double cosi_radar ;
171 
172  // Computing the ray incidence at impact if needed
173  if (penalize_indicence){
174  cosi_radar = vtkMath::Dot(n.colptr(0),target_to_radar_dir_body_frame.colptr(0));
175  }
176  else{
177  cosi_radar = 1;
178  }
179 
180  // A maximum of N points are sampled from this facet
181  // #pragma omp parallel for
182  for (int i = 0; i < N_samples; ++i){
183 
184  // A random origin point is uniformly drawn from this facet
185  arma::vec random = arma::randu<arma::vec>(2);
186  double u = random(0);
187  double v = random(1);
188  arma::vec origin = (1 - std::sqrt(u)) * P0 + std::sqrt(u) * ( 1 - v ) * P1 + std::sqrt(u) * v * P2 ;
189 
190  arma::vec point_above_surface = origin + 3 * tol * n; // the origin of the ray is moved 3*tol above the surface
191 
192  bool has_intersected = this -> check_line_for_intersect(body_index,point_above_surface,radar_pos,BN_dcms_vec,positions_vec,tol);
193 
194  // If this point was not obscured, the return is weighed by the incidence on the inbound and outbout rays
195 
196  if (!has_intersected){
197 
198  // origin is the sampled point, expressed in the body reference frame
199  // Its actual position should be in the inertial frame, accounting for the assigned position of the body
200 
201  arma::vec origin_cm = BN_dcms_vec[body_index].t() * (origin - this -> center_of_mass_vec[body_index]);
202  arma::vec origin_inertial = origin_cm + positions_vec[body_index];
203 
204  // Range
205  double range = arma::norm(origin_inertial - radar_pos);
206 
207  // The velocity at the impact point is a combination of the orbital and rotational velocities
208  arma::vec velocity = velocities_vec[body_index] + arma::cross(omegas_vec[body_index],origin_cm);
209 
210  double range_rate = arma::dot(origin_inertial - radar_pos,velocity) / range;
211 
212  std::array<double, 3> measurement = {{range,range_rate,std::pow(cosi_radar,2)}};
213  measurements.push_back(measurement);
214  }
215 
216 
217  }
218  }
219 }
220 
221 measurements_sequence.push_back(measurements);

225 void SBGATObsRadar::BinObservations(
226  const SBGATRadarObsSequence & measurements_sequence,
227  const double & r_bin,
228  const double & rr_bin){
229 
230  // The container holding the images is pre-allocated
231  this -> images.clear();
232  this -> max_value = -1;
233 
234  for (unsigned int i = 0; i < measurements_sequence.size(); ++i){
235 
236  auto measurements = measurements_sequence[i];
237 
238  this -> images.push_back(vtkSmartPointer<vtkImageData>::New());
239 
240  // Using arma::vec to get statistics
241  arma::vec ranges(measurements.size());
242  arma::vec range_rates(measurements.size());
243 
244  for (unsigned int i = 0; i < measurements.size(); ++i){
245  ranges(i) = measurements[i][0];
246  range_rates(i) = measurements[i][1];
247  }
248 
249  // Subtracting the mean range
250  ranges -= arma::mean(ranges);
251 
252  // The extent of the data is extracted
253  double r_extent = std::abs(ranges.max() - ranges.min()) * this -> scaleFactor;
254  double rr_extent = std::abs(range_rates.max() - range_rates.min()) * this -> scaleFactor;
255 
256 
257  // The bin counts are determined from the specified bin sizes and data extent
258  int n_bin_r = (int)(r_extent / r_bin);
259  int n_bin_rr = (int)(rr_extent / rr_bin);
260 
261  // Checking if one dimension is "empty" (i.e has zero bins)
262  if (n_bin_r == 0 || n_bin_rr == 0){
263  throw(std::runtime_error("The prescribed bin sizes yielded " + std::to_string(n_bin_r) + " range bins and " + std::to_string(n_bin_rr) + " range-rate bins. "));
264  }
265 
266  // The image is formed by "binning in" the measurements
267 
268  this -> images[i] -> SetExtent(0, n_bin_rr - 1, 0, n_bin_r - 1, 0, 0);
269  this -> images[i] -> AllocateScalars(VTK_DOUBLE, 1);
270 
271  double *dPtr = static_cast<double *>(this -> images[i]->GetScalarPointer(0, 0, 0));
272 
273  // The image is initialized
274  #pragma omp parallel for
275  for (int row = 0; row < n_bin_r; ++row){
276  for (int col = 0; col < n_bin_rr; ++col){
277  int k = col + row * n_bin_rr;
278  dPtr[k] = 0;
279  }
280  }
281 
282 
283  double max_range = ranges.max();
284  double max_range_rate = range_rates.max();
285 
286  // The histogram is built
287  #pragma omp parallel for
288  for (unsigned int mes = 0; mes < measurements.size(); ++mes){
289 
290  // Flipping the image
291  int row = int( ( - ranges(mes) + max_range)/ r_bin );
292  int col = int( ( - range_rates(mes) + max_range_rate )/ rr_bin );
293 
294  // The last bin is inclusive
295  if (row == n_bin_r){
296  --row;
297  }
298  if (col == n_bin_rr){
299  --col;
300  }
301 
302  int k = col + row * n_bin_rr;
303 
304  // Adding the cosine of the incidence
305  // If the incidence is 0 then measurements[mes][2] == 1.
306  // if the incidence is 90 deg then measurements[mes][2] == 0
307  // if the incidence is not used then measurements[mes][2] == 1 always
308  dPtr[k] += measurements[mes][2];
309 
310  }
311 
312  // The maximum image value is extracted
313  vtkDataArray * scalars = this -> images[i] -> GetPointData() -> GetScalars();
314 
315  for (vtkIdType tupleIdx = 0; tupleIdx < scalars -> GetNumberOfTuples(); ++tupleIdx){
316  this -> max_value = std::max(max_value,scalars -> GetTuple1(tupleIdx));
317  }
318 
319  }
