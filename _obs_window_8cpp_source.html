216 void ObsWindow::add_state_history(ShapePropertiesWidget * shape_properties_widget,
217  const std::vector<double> & imaging_times,
218  std::vector< std::vector<arma::vec> > & positions_vec,
219  std::vector< std::vector<arma::vec> > & velocities_vec,

227  mass_properties -> SetInputData(shape_data[primary_name] -> get_polydata());
228 
229  mass_properties -> Update();
230 
231 
232 
233  std::vector<arma::vec> position_from_file, attitude_from_file;
234  std::vector<double> position_time_from_file, attitude_time_from_file;
235 
236 
237  if (shape_properties_widget -> position_from_file_button -> isChecked()){
238  // This is where the text file should be parsed
239  // The in the file should be
240  // time (s) -- x_pos (m)-- y_pos (m) -- z_pos (m)-- x_dot (m) -- y_dot (m) -- z_dot
241  ObsWindow::load_state_from_file(
242  shape_properties_widget -> get_position_state_file_path(),
243  position_from_file,
244  position_time_from_file);
245  }
246 
247  if (shape_properties_widget -> attitude_from_file_button -> isChecked()){
248  // This is where the text file should be parsed
249  // time (s) -- sigma_1 -- sigma_2 -- sigma_3 -- omega_x -- omega_y -- omega_z
250  ObsWindow::load_state_from_file(shape_properties_widget -> get_attitude_state_file_path(),

252 
253  }
254 
255  for (int i = 0; i < this -> N_images_sbox -> value(); ++i){
256 
257  double time = imaging_times[i];
258 
259  if (shape_properties_widget -> position_from_keplerian_button -> isChecked()){
260 
261  // If the considered shape is a primary, then its position state should be zero (both in position
262  // and velocity)
263 
264  if (shape_properties_widget -> isPrimary()){
265  positions_vec[i].push_back(arma::zeros<arma::vec>(3));
266  velocities_vec[i].push_back(arma::zeros<arma::vec>(3));
267  }

270  double mu = arma::datum::G * shape_properties_widget -> get_density() * mass_properties -> GetVolume();
271 
272  OC::KepState kep_state(elements,mu);
273  OC::CartState cart_state = kep_state.convert_to_cart(time);
274  positions_vec[i].push_back(cart_state.get_position_vector());
275  velocities_vec[i].push_back(cart_state.get_velocity_vector());
276  }
277 
278  } else {
279 
280  arma::vec position_state = ObsWindow::interpolate(time,position_from_file,position_time_from_file,false);
281 
282  positions_vec[i].push_back(position_state.rows(0,2));
283  velocities_vec[i].push_back(position_state.rows(3,5));
284 
285  }
286 
287  if (shape_properties_widget -> attitude_from_simple_spin_button -> isChecked()){
288  double period = shape_properties_widget -> get_period();
289  double w = 2 * arma::datum::pi / period;
290  double phi = w * time;
291  phi = phi - int(phi / (2 * arma::datum::pi) ) * (2 * arma::datum::pi);
292 
293  arma::vec spin = shape_properties_widget -> get_spin();
294  arma::vec mrp;
295 
296  if (phi > arma::datum::pi){
297  phi = 2 * arma::datum::pi - phi;
298  spin = -spin;
299  }
300

302  arma::vec omega = w * spin;
303  mrps_vec[i].push_back(mrp);
304  omegas_vec[i].push_back(omega);
305 
306  } else{
307

309 
310  mrps_vec[i].push_back(attitude_state.rows(0,2));
311  omegas_vec[i].push_back(attitude_state.rows(3,5));
312 
313  }

335 arma::vec ObsWindow::interpolate(const double & time,
336  const std::vector<arma::vec> & state_from_file,
337  const std::vector<double> & state_time_from_file,
338  bool is_attitude_state){
339 
340 
341  int time_index = -1;
342  double t0,t1;
343  arma::vec f0,f1;
344 
345  for (unsigned int i = 0; i + 1 < state_time_from_file.size(); ++i){
346 
347  if (state_time_from_file[i] < time && state_time_from_file[i + 1] >= time){
348  time_index = i;
349  t0 = state_time_from_file[i];
350  t1 = state_time_from_file[i + 1];
351  f0 = state_from_file[i];
352  f1 = state_from_file[i+1];
353 
354  break;
355  }
356 
357  }
358 
359  if (time_index < 0){
360  throw (std::runtime_error("The prescribed time does not fall within the interpolation time table"));
361  }
362 
363  // If we are dealing with an attitude state, must switch MRP if need be
364  // switching is detected if two consecutive MRPs yield a relative principal rotation vector of more than
365  // 160 degrees in magnitude (a switching over the sigma^Tsigma = 1 surface brings exactly 180 deg of separation)
366  if (is_attitude_state){
367 
368  double prv_relative_norm = arma::norm(RBK::dcm_to_prv(RBK::mrp_to_dcm(f0.rows(0,2))* RBK::mrp_to_dcm(-f1.rows(0,2))));
369 
370  if (prv_relative_norm * 180./arma::datum::pi > 90 ){
371  f0.rows(0,2) = RBK::shadow_mrp(f0.rows(0,2),true);
372  }
373 
374 
375  }
376 
377 
378  arma::vec interpolant = (time - t0)/(t1 - t0) * (f1 - f0) + f0;
379 
380 
381  return interpolant;
382
