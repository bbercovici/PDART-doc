527 void Mainwindow::add_shape() {
528 
529  QString fileName = QFileDialog::getOpenFileName(this,tr("Load shape"), "~/", tr("Wavefront file (*.obj)"));
530 
531  if (fileName.isEmpty() == false) {
532 
533  QStringList items_length_unit;
534  items_length_unit << tr("Meters") << tr("Kilometers");
535  bool ok;
536  QString length_unit = QInputDialog::getItem(this, tr("Length Unit Of Loaded Shape Model:"),tr("Shape Units:"), items_length_unit, 0, false, &ok);
537 
538  if (!ok){
539  return;
540  }
541 
542  QMessageBox::StandardButton enforce_centering_aligment = QMessageBox::question(this, "Shape Alignment", "Force centering on barycenter and principal axes alignment?", QMessageBox::No|QMessageBox::Yes);
543  double scaling_factor;
544  length_unit == "Meters" ? scaling_factor = 1 : scaling_factor = 1000;
545 
546  std::stringstream ss;
547  ss.str(std::string());
548 
549  std::string opening_line = "### Loading shape ###";
550  this -> log_console -> appendPlainText(QString::fromStdString(opening_line));
551  this -> log_console -> appendPlainText(QString::fromStdString("- Loading shape from ") + fileName);
552 
553  std::chrono::time_point<std::chrono::system_clock> start, end;
554 
555  start = std::chrono::system_clock::now();
556 
557  // The name of the shape model is extracted from the path
558  int dot_index = fileName.lastIndexOf(".");
559  int slash_index = fileName.lastIndexOf("/");
560  std::string name = (fileName.toStdString()).substr(slash_index + 1 , dot_index - slash_index - 1);
561  std::string basic_name = name;
562 
563  // A new ModelDataWrapper is created and stored under the name of the shape model
564  std::shared_ptr<ModelDataWrapper> model_data = std::make_shared<ModelDataWrapper>();
565 
566  // Reading
567  vtkNew<vtkOBJReader> reader;
568  reader -> SetFileName(fileName.toStdString().c_str());
569 
570  reader -> Update();
571 
572 
573  // Scaling
574  vtkSmartPointer<vtkTransform> transform = vtkSmartPointer<vtkTransform>::New();
575  transform -> Scale(scaling_factor,scaling_factor,scaling_factor);
576 
577  vtkSmartPointer<vtkTransformPolyDataFilter> transformFilter =vtkSmartPointer<vtkTransformPolyDataFilter>::New();
578  transformFilter -> SetInputConnection(reader -> GetOutputPort());
579  transformFilter -> SetTransform(transform);
580  transformFilter -> SetOutputPointsPrecision ( vtkAlgorithm::DesiredOutputPrecision::DOUBLE_PRECISION );
581 
582  transformFilter -> Update();
583 
584 
585 
586  vtkSmartPointer<vtkIdFilter> idFilter =
587  vtkSmartPointer<vtkIdFilter>::New();
588  idFilter -> SetInputConnection(transformFilter->GetOutputPort());
589  idFilter -> SetIdsArrayName("OriginalIds");
590  idFilter -> Update();
591 
592  vtkSmartPointer<vtkDataSetSurfaceFilter> surfaceFilter = vtkSmartPointer<vtkDataSetSurfaceFilter>::New();
593  surfaceFilter -> SetInputConnection(idFilter -> GetOutputPort());
594  surfaceFilter -> Update();
595 
596 
597 
598  vtkSmartPointer<vtkPolyDataNormals> normals =vtkSmartPointer<vtkPolyDataNormals>::New();
599  normals -> SetInputConnection(surfaceFilter-> GetOutputPort());
600  normals -> SplittingOff();
601  normals -> ConsistencyOn();
602  normals -> ComputeCellNormalsOff();
603  normals -> ComputePointNormalsOn();
604  normals -> Update();
605 
606  // Create a PolyData
607  vtkSmartPointer<vtkPolyData> polygonPolyData = normals -> GetOutput();
608 
609  // The camera is moved to be adjusted to the new shape
610  vtkSmartPointer<SBGATMassProperties> center_of_mass_filter =
611  vtkSmartPointer<SBGATMassProperties>::New();
612 
613  center_of_mass_filter -> SetInputData(polygonPolyData);
614  center_of_mass_filter -> Update();
615 
616  this -> renderer -> GetActiveCamera() -> SetPosition(0, 0, 10 * std::cbrt(3./4. / arma::datum::pi * center_of_mass_filter -> GetVolume()) );
617 
618  // Create a mapper and actor
619  vtkSmartPointer<vtkPolyDataMapper> mapper = vtkSmartPointer<vtkPolyDataMapper>::New();
620 
621  mapper -> SetInputData(polygonPolyData);
622 
623  mapper -> ScalarVisibilityOff();
624 
625  vtkSmartPointer<vtkActor> actor =vtkSmartPointer<vtkActor>::New();
626  actor -> SetMapper(mapper);
627 
628  // Visualize
629  this -> renderer -> AddActor(actor);
630 
631 
632  // Create the tree
633  vtkSmartPointer<vtkKdTreePointLocator> tree = vtkSmartPointer<vtkKdTreePointLocator>::New();
634  tree -> SetDataSet(polygonPolyData);
635  tree -> BuildLocator();
636 
637  // Store
638  model_data -> set_polydata(polygonPolyData);
639  model_data -> set_actor(actor);
