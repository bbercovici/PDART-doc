412 void SBGATFilterUQ::TakeSlice(int axis,
413  std::vector<std::vector<arma::vec> > & lines,
414  const double & c) const{
415 
416  arma::vec n_plane = {0,0,0};
417  n_plane(axis) = 1;
418 
419 
420  arma::mat::fixed<3,2> T = {
421  {1,0},
422  {0,1},
423  {-1,-1}
424  };
425 
426  arma::vec::fixed<3> e3 = {0,0,1};
427  arma::mat::fixed<3,3> C;
428 
429  // Each surface element is "sliced"
430  for (int el = 0; el < this -> model -> GetN_facets(); ++el){
431 
432 
433  double rf0[3];
434  double rf1[3];
435  double rf2[3];
436 
437  int p0,p1,p2;
438  this -> model -> GetIndicesVerticesInFacet(el,p0,p1,p2);
439  this -> model -> GetVerticesInFacet(el,rf0,rf1,rf2);
440 
441  C.col(0) = arma::vec({rf0[0],rf0[1],rf0[2]});
442  C.col(1) = arma::vec({rf1[0],rf1[1],rf1[2]});
443  C.col(2) = arma::vec({rf2[0],rf2[1],rf2[2]});
444 
445  arma::rowvec M = n_plane.t() * C * T;
446  double e = c - arma::dot(n_plane,C * e3);
447  arma::vec intersect;
448 
449  std::vector <arma::vec> intersects;
450 
451  // Looking for an intersect along the u = 0 edge
452  if (std::abs(M(1)) > 1e-16){
453  double v_intersect = e / M(1);
454  if (v_intersect >= 0 && v_intersect <= 1 ){
455  arma::vec Y = {0,v_intersect};
456 
457  intersect = C * (T * Y + e3);
458  intersects.push_back(intersect);
459  }
460  }
461 
462  // Looking for an intersect along the v = 0 edge
463  if (std::abs(M(0)) > 1e-16){
464  double u_intersect = e / M(0);
465  if (u_intersect >= 0 && u_intersect <= 1 ){
466  arma::vec Y = {u_intersect,0};
467 
468  intersect = C * (T * Y + e3);
469  intersects.push_back(intersect);
470  }
471  }
472 
473  // Looking for an intersect along the w = 0 edge
474  // using u as the parameter
475 
476  if (std::abs(M(0) - M(1)) > 1e-16 && intersects.size() < 2){
477  double u_intersect = (e - M(1)) / (M(0) - M(1));
478  if (u_intersect >= 0 && u_intersect <= 1 ){
479  arma::vec Y = {u_intersect,1 - u_intersect};
480 
481  intersect = C * (T * Y + e3);
482  intersects.push_back(intersect);
483  }
484  }
485 
486  if (intersects.size() == 2){
487  lines.push_back(intersects);
488  }
489 
490 
491 
492  }
