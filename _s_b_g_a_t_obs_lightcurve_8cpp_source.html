161 void SBGATObsLightcurve::reverse_ray_trace(std::array<double, 2> & measurements_temp,
162  const std::vector<std::vector<int> > & facets_in_view,
163  const arma::vec & sun_dir,
164  const arma::vec & observer_dir,
165  const int N,
166  const bool penalize_indicence,
167  const std::vector<arma::mat> & BN_dcms_vec,
168  const std::vector<arma::vec> & positions_vec){
169 
170 
171 
172 
173  // The sun is positionned with respect to the primary
174  arma::vec observer_pos = this -> center_of_mass_vec[0] + this -> polydata_vec[0] -> GetLength() * 1E6 * observer_dir;
175  arma::vec sun_pos = this -> center_of_mass_vec[0] + this -> polydata_vec[0] -> GetLength() * 1E6 * sun_dir;
176 
177  // Ray-tracing tolerance
178  double tol = this -> polydata_vec[0] -> GetLength()/1E6;
179 
180 
181 
182 
183 
184  for (int body_index = 0; body_index < this -> number_of_bodies; ++body_index){
185 
186  vtkPolyData * input = this -> polydata_vec[body_index];
187 
188  vtkSmartPointer<vtkIdList> ptIds = vtkSmartPointer<vtkIdList>::New();
189  ptIds -> Allocate(VTK_CELL_SIZE);
190 
191  arma::vec::fixed<3> target_to_sun_dir_body_frame = BN_dcms_vec[body_index] * sun_dir;
192  arma::vec::fixed<3> target_to_observer_dir_body_frame = BN_dcms_vec[body_index] * observer_dir;
193 
194 
195  // The kept facets are then sampled and reverse ray-traced
196  for (unsigned int facet_index = 0; facet_index != facets_in_view[body_index].size(); ++facet_index){
197 
198  double p0[3];
199  double p1[3];
200  double p2[3];
201 
202  input -> GetCellPoints(facets_in_view[body_index][facet_index],ptIds);
203  input -> GetPoint(ptIds->GetId(0), p0);
204  input -> GetPoint(ptIds->GetId(1), p1);
205  input -> GetPoint(ptIds->GetId(2), p2);
206 
207  arma::vec::fixed<3> P0 = {p0[0],p0[1],p0[2]};
208  arma::vec::fixed<3> P1 = {p1[0],p1[1],p1[2]};
209  arma::vec::fixed<3> P2 = {p2[0],p2[1],p2[2]};
210 
211  arma::vec::fixed<3> n = arma::cross(P1 - P0, P2 - P0);
212 
213  // The number of points sampled from this facet is determined based on
214  // the relative size of this facet compared to the largest one in all the considered shapes
215  int N_samples = int( N * arma::norm(n /2) / this -> min_area);
216 
217  // only need unit normal vector from here
218  n = arma::normalise(arma::cross(P1 - P0, P2 - P0));
219 
220  double cosi_sun,cosi_obs ;
221 
222  // Computing the ray incidence at impact if needed
223  if (penalize_indicence){
224  cosi_sun = vtkMath::Dot(n.colptr(0),target_to_sun_dir_body_frame.colptr(0));
225  cosi_obs = vtkMath::Dot(n.colptr(0),target_to_observer_dir_body_frame.colptr(0));
226  }
227  else{
228  cosi_sun = 1;
229  cosi_obs = 1;
230  }
231 
232  // A maximum of N points are sampled from this facet
233  // #pragma omp parallel for
234  for (int i = 0; i < N_samples; ++i){
235 
236 
237  // A random origin point is uniformly drawn from this facet
238  arma::vec random = arma::randu<arma::vec>(2);
239  double u = random(0);
240  double v = random(1);
241  arma::vec::fixed<3> origin = (1 - std::sqrt(u)) * P0 + std::sqrt(u) * ( 1 - v ) * P1 + std::sqrt(u) * v * P2 ;
242 
243  // Derived points are expressed in the body reference frame
244  arma::vec point_above_surface = origin + 3 * tol * n; // the origin of the ray is moved 3*tol above the surface
245 
246  bool has_intersected = (this -> check_line_for_intersect(body_index,point_above_surface,sun_pos,BN_dcms_vec,positions_vec,tol)
247  + this -> check_line_for_intersect(body_index,point_above_surface,observer_pos,BN_dcms_vec,positions_vec,tol));
248 
249  // If this point was not obscured, the return is weighed by the incidence on the inbound and outbout rays
250  if (!has_intersected){
251  measurements_temp[1] += cosi_sun * cosi_obs;
252  }
253 
254 
255  }
256  }
257 
258 }
