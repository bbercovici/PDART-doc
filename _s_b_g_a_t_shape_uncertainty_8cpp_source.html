184 void SBGATShapeUncertainty::ComputeInertiaStatisticsMC(int N_samples,double l,double sigma,std::string output_dir){
185 
186  this -> Update();
187  this -> shape_model_bezier -> compute_point_covariances(std::pow(sigma,2),l);
188  this -> shape_model_bezier -> compute_shape_covariance_sqrt();
189 
190  arma::vec results_volume;
191  arma::mat results_cm,results_inertia,results_moments,results_mrp,results_dims;
192 
193  this -> shape_model_bezier -> run_monte_carlo(N_samples,
194  results_volume,
195  results_cm,
196  results_inertia,
197  results_moments,
198  results_mrp,
199  results_dims,
200  output_dir);
201 
202 
203  arma::vec results_cm_mean = arma::mean(results_cm,1);
204  arma::vec results_inertia_mean = arma::mean(results_inertia,1);
205  arma::vec results_moments_mean = arma::mean(results_moments,1);
206  arma::vec results_dims_mean = arma::mean(results_dims,1);
207  arma::vec results_mrp_mean = arma::mean(results_mrp,1);
208 
209  arma::mat cov_cm_mc = arma::zeros(3,3);
210  arma::mat cov_inertia_mc = arma::zeros(6,6);
211  arma::mat cov_moments_mc = arma::zeros(4,4);
212  arma::mat cov_dims_mc = arma::zeros<arma::mat>(3,3);
213  arma::mat cov_mrp_mc = arma::zeros(3,3);
214 
215  for (unsigned int i = 0; i < results_cm.n_cols; ++i){
216 
217  cov_cm_mc += (results_cm.col(i) - results_cm_mean) * (results_cm.col(i) - results_cm_mean).t();
218  cov_inertia_mc += (results_inertia.col(i) - results_inertia_mean) * (results_inertia.col(i) - results_inertia_mean).t();
219  cov_moments_mc += (results_moments.col(i) - results_moments_mean) * (results_moments.col(i) - results_moments_mean).t();
220  cov_dims_mc += (results_dims.col(i) - results_dims_mean) * (results_dims.col(i) - results_dims_mean).t();
221  cov_mrp_mc += (RBK::dcm_to_mrp(RBK::mrp_to_dcm(results_mrp.col(i))*RBK::mrp_to_dcm(results_mrp_mean).t())) * (RBK::dcm_to_mrp(RBK::mrp_to_dcm(results_mrp.col(i))*RBK::mrp_to_dcm(results_mrp_mean).t())).t();
222 
223  }
224 
225  cov_cm_mc *= 1./(results_cm.n_cols-1);
226  cov_inertia_mc *= 1./(results_inertia.n_cols-1);
227  cov_moments_mc *= 1./(results_moments.n_cols-1);
228  cov_dims_mc *= 1./(results_dims.n_cols-1);
229  cov_mrp_mc *= 1./(results_mrp.n_cols-1);
230 
231  this -> volume_variance = std::pow(arma::stddev(results_volume,1),2);
232  this -> center_of_mass_covariance = cov_cm_mc;
233  this -> inertia_tensor_parametrization_covariance = cov_inertia_mc;
234  this -> principal_dimensions_covariance = cov_dims_mc;
235  this -> principal_moments_covariance = cov_moments_mc;
236 
237  this -> mrp_covariance = cov_mrp_mc;
