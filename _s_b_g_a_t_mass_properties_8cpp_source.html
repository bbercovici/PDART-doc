146  sum_surface[1] = 0;
147  sum_surface[2] = 0;

152  arma::vec::fixed<3> bbox_min = {this -> bounds[0],this -> bounds[2],this -> bounds[4]};
153  arma::vec::fixed<3> bbox_max = {this -> bounds[1],this -> bounds[3],this -> bounds[5]};
154  l = arma::norm(bbox_max - bbox_min);
155 
156  for ( idx = 0; idx < 3 ; idx++ ){
157  munc[idx] = 0.0;
158  vol[idx] = 0.0;
159  kxyz[idx] = 0.0;
160  }
161 
162  for (cellId=0; cellId < numCells; cellId++){
163  if ( input->GetCellType(cellId) != VTK_TRIANGLE){
164  vtkWarningMacro(<< "Input data type must be VTK_TRIANGLE not "<< input->GetCellType(cellId));
165  continue;
166  }
167  input->GetCellPoints(cellId,ptIds);
168  numIds = ptIds->GetNumberOfIds();
169  assert(numIds == 3);
170 
171  // store current vertex (x,y,z) coordinates ...
172  // Note that the coordinates are normalized!
173  for (idx=0; idx < numIds; idx++){
174  input->GetPoint(ptIds->GetId(idx), p);
175  x[idx] = p[0] / l; y[idx] = p[1] / l; z[idx] = p[2] / l;
176  }
177 
178  // get i j k vectors ...
179  //
180  i[0] = ( x[1] - x[0]); j[0] = (y[1] - y[0]); k[0] = (z[1] - z[0]);
181  i[1] = ( x[2] - x[0]); j[1] = (y[2] - y[0]); k[1] = (z[2] - z[0]);
182  i[2] = ( x[2] - x[1]); j[2] = (y[2] - y[1]); k[2] = (z[2] - z[1]);
183 
184  // cross product between two vectors, to determine normal vector
185  //
186  u[0] = ( j[0] * k[1] - k[0] * j[1]);
187  u[1] = ( k[0] * i[1] - i[0] * k[1]);
188  u[2] = ( i[0] * j[1] - j[0] * i[1]);
189 
190  // normalize normal vector to 1
191  //
192  length = sqrt( u[0]*u[0] + u[1]*u[1] + u[2]*u[2]);
193  if ( length != 0.0)
194  {
195  u[0] /= length;
196  u[1] /= length;
197  u[2] /= length;
198  }
199  else
200  {
201  u[0] = u[1] = u[2] = 0.0;
202  }
203 
204  // determine max unit normal component...
205  //
206  absu[0] = fabs(u[0]); absu[1] = fabs(u[1]); absu[2] = fabs(u[2]);
207 
208  if (( absu[0] > absu[1]) && ( absu[0] > absu[2]) )
209  {
210  munc[0]++;
211  }
212  else if (( absu[1] > absu[0]) && ( absu[1] > absu[2]) )
213  {
214  munc[1]++;
215  }
216  else if (( absu[2] > absu[0]) && ( absu[2] > absu[1]) )
217  {
218  munc[2]++;
219  }
220  else if (( absu[0] == absu[1])&& ( absu[0] == absu[2]))
221  {
222  wxyz++;
223  }
224  else if (( absu[0] == absu[1])&& ( absu[0] > absu[2]) )
225  {
226  wxy++;
227  }
228  else if (( absu[0] == absu[2])&& ( absu[0] > absu[1]) )
229  {
230  wxz++;
231  }
232  else if (( absu[1] == absu[2])&& ( absu[0] < absu[2]) )
233  {
234  wyz++;
235  }
236  else
237  {
238  vtkErrorMacro( << "Unpredicted situation...!" );
239  return 1;
240  }
241 
242  // This is reduced to ...
243  //
244  ii[0] = i[0] * i[0]; ii[1] = i[1] * i[1]; ii[2] = i[2] * i[2];
245  jj[0] = j[0] * j[0]; jj[1] = j[1] * j[1]; jj[2] = j[2] * j[2];
246  kk[0] = k[0] * k[0]; kk[1] = k[1] * k[1]; kk[2] = k[2] * k[2];
247 
248  // area of a triangle...
249  //
250  a = sqrt(ii[1] + jj[1] + kk[1]);
251  b = sqrt(ii[0] + jj[0] + kk[0]);
252  c = sqrt(ii[2] + jj[2] + kk[2]);
253  s = 0.5 * (a + b + c);
254  area = sqrt( fabs(s*(s-a)*(s-b)*(s-c)));
255  surfacearea += area;
256 
257  average_surface += area / numCells;
258 
259 
260 
261  if( area < mincellarea )
262  {
263  mincellarea = area;
264  }
265  if( area > maxcellarea )
266  {
267  maxcellarea = area;
268  }
269 
270  // volume elements ...
271  //
272  zavg = (z[0] + z[1] + z[2]) / 3.0;
273  yavg = (y[0] + y[1] + y[2]) / 3.0;
274  xavg = (x[0] + x[1] + x[2]) / 3.0;
275 
276  vol[2] += (area * u[2] * zavg);
277  vol[1] += (area * u[1] * yavg);
278  vol[0] += (area * u[0] * xavg);
279 
280  // V = (z1+z2+z3)(x1y2-x2y1+x2y3-x3y2+x3y1-x1y3)/6
281  // Volume under triangle is projected area of the triangle times
282  // the average of the three z values
283  vtkMath::Cross(x,y,xp);
284  volumeproj += zavg * (xp[0]+xp[1]+xp[2]) / 2;
285 
286  // Center of mass
287  // See "Inertia of Any Polyhedron" by Anthony R. Dobrovolskis, Icarus 124, 698–704 (1996) Article No. 0243
288  dv = 1. / 6. * (x[1] * ( ( y[1] - y[0] ) * ( z[2] - z[0] ) - ( z[1] - z[0] ) * ( y[2] - y[0])) +
289  y[1] * ( ( z[1] - z[0] ) * ( x[2] - x[0] ) - ( x[1] - x[0] ) * ( z[2] - z[0] ) ) +
290  z[1] * ( ( x[1] - x[0] ) * ( y[2] - y[0] ) - ( y[1] - y[0] ) * ( x[2] - x[0] )));
291 
292  cx += dv * (x[0] + x[1] + x[2])/4;
293  cy += dv * (y[0] + y[1] + y[2])/4;
294  cz += dv * (z[0] + z[1] + z[2])/4;
295 
296 
297  // Inertia tensor
298  // See "Inertia of Any Polyhedron" by Anthony R. Dobrovolskis, Icarus 124, 698–704 (1996) Article No. 0243
299 
300  P_xx += dv / 20 * (2 * x[0] * x[0]
301  + 2 * x[1] * x[1]
302  + 2 * x[2] * x[2]
303  + x[0] * x[1]
304  + x[0] * x[1]
305  + x[0] * x[2]
306  + x[0] * x[2]
307  + x[1] * x[2]
308  + x[1] * x[2]);
309 
310 
311  P_yy += dv / 20 * (2 * y[0] * y[0]
312  + 2 * y[1] * y[1]
313  + 2 * y[2] * y[2]
314  + y[0] * y[1]
315  + y[0] * y[1]
316  + y[0] * y[2]
317  + y[0] * y[2]
318  + y[1] * y[2]
319  + y[1] * y[2]);
320 
321  P_zz += dv / 20 * (2 * z[0] * z[0]
322  + 2 * z[1] * z[1]
323  + 2 * z[2] * z[2]
324  + z[0] * z[1]
325  + z[0] * z[1]
326  + z[0] * z[2]
327  + z[0] * z[2]
328  + z[1] * z[2]
329  + z[1] * z[2]);
330 
331  P_xy += dv / 20 * (2 * x[0] * y[0]
332  + 2 * x[1] * y[1]
333  + 2 * x[2] * y[2]
334  + x[0] * y[1]
335  + y[0] * x[1]
336  + x[0] * y[2]
337  + y[0] * x[2]
338  + x[1] * y[2]
339  + y[1] * x[2]);
340 
341  P_xz += dv / 20 * (2 * x[0] * z[0]
342  + 2 * x[1] * z[1]
343  + 2 * x[2] * z[2]
344  + x[0] * z[1]
345  + z[0] * x[1]
346  + x[0] * z[2]
347  + z[0] * x[2]
348  + x[1] * z[2]
349  + z[1] * x[2]);
350 
351  P_yz += dv / 20 * (2 * y[0] * z[0]
352  + 2 * y[1] * z[1]
353  + 2 * y[2] * z[2]
354  + y[0] * z[1]
355  + z[0] * y[1]
356  + y[0] * z[2]
357  + z[0] * y[2]
358  + y[1] * z[2]
359  + z[1] * y[2]);
360 
361  // Sum of oriented surface
362  vtkMath::MultiplyScalar(u,area);
363  vtkMath::Add(u,sum_surface,sum_surface);
364 
365  }

367  // Surface Area ...
368  //
369  this->SurfaceArea = surfacearea* std::pow(l,2);
370  this->MinCellArea = mincellarea* std::pow(l,2);
371  this->MaxCellArea = maxcellarea* std::pow(l,2);
372 
373  // Weighting factors in Discrete Divergence theorem for volume calculation.
374  //
375  kxyz[0] = (munc[0] + (wxyz/3.0) + ((wxy+wxz)/2.0)) /numCells;
376  kxyz[1] = (munc[1] + (wxyz/3.0) + ((wxy+wyz)/2.0)) /numCells;
377  kxyz[2] = (munc[2] + (wxyz/3.0) + ((wxz+wyz)/2.0)) /numCells;
378  this->VolumeX = vol[0] * std::pow(l,3) ;
379  this->VolumeY = vol[1] * std::pow(l,3) ;
380  this->VolumeZ = vol[2] * std::pow(l,3) ;
381  this->Kx = kxyz[0];
382  this->Ky = kxyz[1];
383  this->Kz = kxyz[2];
384  this->Volume = (kxyz[0] * vol[0] + kxyz[1] * vol[1] + kxyz[2] * vol[2])* std::pow(l,3);
385  this->Volume = fabs(this->Volume);
386  this->VolumeProjected = volumeproj * std::pow(l,3);
387  this->NormalizedShapeIndex =
388  (sqrt(surfacearea)/std::cbrt(this->Volume))/2.199085233;
389 
390 
391  // Center of mass
392  arma::vec::fixed<3> com_ = {cx,cy,cz};
393 
394  this -> center_of_mass = l * com_ / this->Volume * std::pow(l,3);
395 
396  // Inertia tensor that was non-dimensionalized by l, not cbrt(V)
397  arma::mat I = {
398  {P_yy + P_zz, -P_xy, -P_xz},
399  { -P_xy, P_xx + P_zz, -P_yz},
400  { -P_xz, -P_yz, P_xx + P_yy}};
401 
402  double L = std::cbrt(this -> Volume) ;
403  this -> inertia_tensor = std::pow(l / L,5) * I - RBK::tilde(this -> center_of_mass / L) * RBK::tilde(this -> center_of_mass / L).t();
404 
405  // The principal axes are extracted
406  arma::vec eig_val;
407  arma::mat eig_vec;
408  arma::eig_sym(eig_val,eig_vec,this -> inertia_tensor);
409 
410 
411  // At this stage, eig_vec is (+/- v0,+/- v1 , +/- v2) where |v0| = |v1| = |v2| = 1
412  // with their corresponding values sorted in ascending order
413  // But it is not guaranteed that det(eig_vec) = +1
414  // The 8 total configurations of the principal axes matrix M that exist are
415  // M1: (v0,v1,v2) == eig_vec
416  // M2: (v0,v1,-v2)
417  // M3: (v0,-v1,v2)
418  // M4: (v0,-v1,-v2)
419  // M5: (-v0,v1,v2)
420  // M6: (-v0,v1,-v2)
421  // M7: (-v0,-v1,v2)
422  // M8: (-v0,-v1,-v2)
423 
424  // Note that M1,M4,M6,M7 and M2,M3,M5,M8 have the same determinant respectively
425 
426  if (arma::det(eig_vec) < 0){
427  eig_vec = - eig_vec;
428  }
429 
430  if(eig_vec(0,0) > 0 && eig_vec(1,1) < 0){
431 
432  eig_vec.col(1) = - eig_vec.col(1);
433  eig_vec.col(2) = - eig_vec.col(2);
434  }
435 
436  else if(eig_vec(0,0) < 0 && eig_vec(2,2) < 0){
437 
438  eig_vec.col(0) = - eig_vec.col(0);
439  eig_vec.col(2) = - eig_vec.col(2);
440  }
441 
442  else if(eig_vec(0,0) < 0 && eig_vec(1,1) < 0){
443 
444  eig_vec.col(0) = - eig_vec.col(0);
445  eig_vec.col(1) = - eig_vec.col(1);
446  }
447 
448  this -> principal_axes = eig_vec;
449 
450 
451  // Closeness of topology given sum of oriented surface
452  if (vtkMath::Norm(sum_surface) / average_surface < 1e-6){
453  this -> IsClosed = true;
454  }
455  else{
456  this -> IsClosed = false;
457  }
458 
459  return 1;
