248 arma::vec::fixed<3> SBGATPolyhedronGravityModel::GetAcceleration(double const * point) const{
249 
250  double point_scaled[3] = {point[0],point[1],point[2]};
251  vtkMath::MultiplyScalar(point_scaled,1./this -> scaleFactor);
252 
253  arma::vec::fixed<3> acc = arma::zeros<arma::vec>(3);
254 
255  // Facet loop
256  #pragma omp parallel for reduction(+:acc)
257  for (vtkIdType facet_index = 0; facet_index < this -> N_facets; ++ facet_index) {
258 
259  double * r0 = this -> vertices[this -> facets[facet_index][0]];
260 
261  double r0m[3];
262 
263  vtkMath::Subtract(r0,point_scaled,r0m);
264 
265  double wf = this -> GetPerformanceFactor( point, facet_index);
266  double * F = this -> facet_dyads[facet_index];
267 
268  acc(0) += wf *( F[0] * r0m[0] + F[1] * r0m[1] + F[2] * r0m[2]);
269  acc(1) += wf *( F[3] * r0m[0] + F[4] * r0m[1] + F[5] * r0m[2]);
270  acc(2) += wf *( F[6] * r0m[0] + F[7] * r0m[1] + F[8] * r0m[2]);
271 
272  }
273 
274  // Edge loop
275  #pragma omp parallel for reduction(+:acc)
276  for (int edge_index = 0; edge_index < this -> N_edges; ++ edge_index) {
277 
278  double * r0 = this -> vertices[this -> edges[edge_index][0]];
279 
280  double r0m[3];
281 
282  vtkMath::Subtract(r0,point_scaled,r0m);
283 
284  double Le = this -> GetLe( point, edge_index);
285 
286  double * E = this -> edge_dyads[edge_index];
287 
288  acc(0) += - Le *( E[0] * r0m[0] + E[1] * r0m[1] + E[2] * r0m[2]);
289  acc(1) += - Le *( E[3] * r0m[0] + E[4] * r0m[1] + E[5] * r0m[2]);
290  acc(2) += - Le *( E[6] * r0m[0] + E[7] * r0m[1] + E[8] * r0m[2]);
291 
292  }
293 
294  return acc * arma::datum::G * this -> density * this -> scaleFactor;
295

308 void SBGATPolyhedronGravityModel::GetPotentialAcceleration(double const * point,double & potential,
309  arma::vec::fixed<3> & acc) const {
310 
311  double pot = 0;
312  double acc_x = 0;
313  double acc_y = 0;
314  double acc_z = 0;
315 
316  double point_scaled[3] = {point[0],point[1],point[2]};
317  vtkMath::MultiplyScalar(point_scaled,1./this -> scaleFactor);
318 
319  // Facet loop
320  #pragma omp parallel for reduction(+:acc_x,acc_y,acc_z,pot)
321  for (vtkIdType facet_index = 0; facet_index < this -> N_facets; ++ facet_index) {
322 
323  double * r0 = this -> vertices[this -> facets[facet_index][0]];
324 
325  double r0m[3];
326 
327  vtkMath::Subtract(r0,point_scaled,r0m);
328 
329  double wf = this -> GetPerformanceFactor( point, facet_index);
330 
331  double * F = this -> facet_dyads[facet_index];
332 
333  double a[3] = {
334  F[0] * r0m[0] + F[1] * r0m[1] + F[2] * r0m[2],
335  F[3] * r0m[0] + F[4] * r0m[1] + F[5] * r0m[2],
336  F[6] * r0m[0] + F[7] * r0m[1] + F[8] * r0m[2]
337  };
338 
339  acc_x += wf * a[0];
340  acc_y += wf * a[1];
341  acc_z += wf * a[2];
342 
343  pot += - wf * vtkMath::Dot(r0m,a);
344 
345 
346  }
347 
348  // Edge loop
349  #pragma omp parallel for reduction(-:acc_x,acc_y,acc_z,pot)
350  for (int edge_index = 0; edge_index < this -> N_edges; ++ edge_index) {
351 
352  double * r0 = this -> vertices[this -> edges[edge_index][0]];
353 
354 
355  double r0m[3];
356 
357  vtkMath::Subtract(r0,point_scaled,r0m);
358 
359 
360  double Le = this -> GetLe( point, edge_index);
361 
362  double * E = this -> edge_dyads[edge_index];
363 
364  double a[3] = {
365  E[0] * r0m[0] + E[1] * r0m[1] + E[2] * r0m[2],
366  E[3] * r0m[0] + E[4] * r0m[1] + E[5] * r0m[2],
367  E[6] * r0m[0] + E[7] * r0m[1] + E[8] * r0m[2]
368  };
369 
370 
371  pot += Le * vtkMath::Dot(r0m,a);
372 
373  acc_x -= Le * a[0];
374  acc_y -= Le * a[1];
375  acc_z -= Le * a[2];
376 
377  }
378 
379  acc(0) = acc_x;
380  acc(1) = acc_y;
381  acc(2) = acc_z;
382 
383  acc *= arma::datum::G * this -> density * this -> scaleFactor ;
384  pot *= 0.5 * arma::datum::G * this -> density * this -> scaleFactor* this -> scaleFactor;
385 
386  potential = pot;
387 
388 }

397 void SBGATPolyhedronGravityModel::GetPotentialAccelerationGravityGradient(double const * point,double & potential,
398  arma::vec::fixed<3> & acc,arma::mat::fixed<3,3> & gravity_gradient_mat) const{
399 
400  double point_scaled[3] = {point[0],point[1],point[2]};
401  vtkMath::MultiplyScalar(point_scaled,1./this -> scaleFactor);
402 
403 
404 
405 
406  double pot = 0;
407  acc = arma::zeros<arma::vec>(3);
408  gravity_gradient_mat = arma::zeros<arma::mat>(3,3);
409 
410  // Facet loop
411  #pragma omp parallel for reduction(+:pot) reduction(+:acc) reduction(+:gravity_gradient_mat)
412  for (vtkIdType facet_index = 0; facet_index < this -> N_facets; ++ facet_index) {
413 
414  double * r0 = this -> vertices[this -> facets[facet_index][0]];
415 
416  double r0m[3];
417 
418  vtkMath::Subtract(r0,point_scaled,r0m);
419 
420  double wf = this -> GetPerformanceFactor( point, facet_index);
421 
422  double * F = this -> facet_dyads[facet_index];
423 
424  double a[3] = {
425  F[0] * r0m[0] + F[1] * r0m[1] + F[2] * r0m[2],
426  F[3] * r0m[0] + F[4] * r0m[1] + F[5] * r0m[2],
427  F[6] * r0m[0] + F[7] * r0m[1] + F[8] * r0m[2]
428  };
429 
430  arma::mat::fixed<3,3> F_arma = {
431  {F[0],F[1],F[2]},
432  {F[3],F[4],F[5]},
433  {F[6],F[7],F[8]}
434  };
435 
436  acc += wf * arma::vec({a[0],a[1],a[2]});
437 
438  pot += - wf * vtkMath::Dot(r0m,a);
439 
440  gravity_gradient_mat += (- F_arma * wf);
441 
442 
443  }
444 
445  // Edge loop
446  #pragma omp parallel for reduction(+:acc) reduction(+:pot) reduction(+:gravity_gradient_mat)
447  for (int edge_index = 0; edge_index < this -> N_edges; ++ edge_index) {
448 
449  double * r0 = this -> vertices[this -> edges[edge_index][0]];
450 
451 
452  double r0m[3];
453 
454  vtkMath::Subtract(r0,point_scaled,r0m);
455 
456  double Le = this -> GetLe( point, edge_index);
457 
458  double * E = this -> edge_dyads[edge_index];
459 
460  double a[3] = {
461  E[0] * r0m[0] + E[1] * r0m[1] + E[2] * r0m[2],
462  E[3] * r0m[0] + E[4] * r0m[1] + E[5] * r0m[2],
463  E[6] * r0m[0] + E[7] * r0m[1] + E[8] * r0m[2]
464  };
465

467  {E[0],E[1],E[2]},
468  {E[3],E[4],E[5]},
469  {E[6],E[7],E[8]}
470  };
471 
472  pot += Le * vtkMath::Dot(r0m,a);
473 
474  acc += arma::vec({a[0],a[1],a[2]}) * (- Le);
475 
476  gravity_gradient_mat += Le * E_arma;
477 
478 
479 
480  }
481 
482  acc *= arma::datum::G * this -> density * this -> scaleFactor ;
483  pot *= 0.5 * arma::datum::G * this -> density * this -> scaleFactor * this -> scaleFactor;
484  gravity_gradient_mat *= arma::datum::G * this -> density;
485 
486  potential = pot;
487 
488 }

540 void SBGATPolyhedronGravityModel::ComputeSurfacePGM(
541  vtkSmartPointer<vtkPolyData> selected_shape,
542  const std::vector<unsigned int> & queried_elements,
543  bool is_in_meters,
544  double density,
545  const arma::vec::fixed<3> & omega,
546  std::vector<double> & slopes,
547  std::vector<double> & inertial_potentials,
548  std::vector<double> & body_fixed_potentials,
549  std::vector<double> & inertial_acc_magnitudes,
550  std::vector<double> & body_fixed_acc_magnitudes){
551 
552  vtkSmartPointer<SBGATPolyhedronGravityModel> pgm_filter = vtkSmartPointer<SBGATPolyhedronGravityModel>::New();
553  pgm_filter -> SetInputData(selected_shape);
554  pgm_filter -> SetDensity(density);
555 
556  vtkSmartPointer<SBGATMassProperties> mass_prop = vtkSmartPointer<SBGATMassProperties>::New();
557  mass_prop -> SetInputData(selected_shape);
558 
559  if (is_in_meters){
560  pgm_filter -> SetScaleMeters();
561  mass_prop -> SetScaleMeters();
562  }
563  else{
564  pgm_filter -> SetScaleKiloMeters();
565  mass_prop -> SetScaleKiloMeters();
566  }
567 
568  pgm_filter -> Update();
569  mass_prop -> Update();
570 
571  slopes.clear();
572  inertial_potentials.clear();
573  body_fixed_potentials.clear();
574  inertial_acc_magnitudes.clear();
575  body_fixed_acc_magnitudes.clear();
576 
577  for (int i = 0; i < static_cast<int>(selected_shape -> GetNumberOfCells()); ++i){
578 
579  slopes.push_back(std::numeric_limits<double>::quiet_NaN());
580  inertial_potentials.push_back(std::numeric_limits<double>::quiet_NaN());
581  body_fixed_potentials.push_back(std::numeric_limits<double>::quiet_NaN());
582  inertial_acc_magnitudes.push_back(std::numeric_limits<double>::quiet_NaN());
583  body_fixed_acc_magnitudes.push_back(std::numeric_limits<double>::quiet_NaN());
584 
585  }
586 
587 
588  arma::vec::fixed<3> com = mass_prop -> GetCenterOfMass();
589 
590  // The queried facets are browsed and their surface PGM evaluated
591  #pragma omp parallel for
592  for ( int e = 0; e < queried_elements.size(); ++e){
593 
594  int el = queried_elements[e];
595 
596  unsigned int cellId = queried_elements.at(el);
597 
598  double p0[3];
599  double p1[3];
600  double p2[3];
601 
602  pgm_filter -> GetVerticesInFacet(el,p0,p1, p2);
603 
604 
605  arma::vec::fixed<3> p0_arma = {p0[0],p0[1],p0[2]};
606  arma::vec::fixed<3> p1_arma = {p1[0],p1[1],p1[2]};
607  arma::vec::fixed<3> p2_arma = {p2[0],p2[1],p2[2]};
608  arma::vec::fixed<3> facet_center = 1./3 * (p0_arma + p1_arma + p2_arma);
609  arma::vec::fixed<3> normal = arma::normalise(arma::cross(p1_arma - p0_arma,p2_arma - p0_arma));
610 
611  double potential,slope;
612  arma::vec::fixed<3> acc,acc_body_fixed;
613 
614  pgm_filter -> GetPotentialAcceleration(facet_center ,potential,acc);
615  acc_body_fixed = acc - arma::cross(omega,arma::cross(omega,facet_center - com));
616 
617  slope = std::acos(arma::dot(-arma::normalise(acc_body_fixed),normal)) * 180./arma::datum::pi;
618 
619  slopes[cellId] = slope;
620  inertial_potentials[cellId] = potential;
621  body_fixed_potentials[cellId] = potential + 0.5 * arma::dot(RBK::tilde(omega) * (facet_center - com),
622  RBK::tilde(omega) * (facet_center - com));
623  inertial_acc_magnitudes[cellId] = arma::norm(acc);
624  body_fixed_acc_magnitudes[cellId] = arma::norm(acc_body_fixed);
625 
626  }
627
