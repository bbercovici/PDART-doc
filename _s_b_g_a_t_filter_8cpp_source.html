83 int SBGATFilter::RequestData(
84  vtkInformation* vtkNotUsed( request ),
85  vtkInformationVector** inputVector,
86  vtkInformationVector* vtkNotUsed( outputVector )){
87  vtkInformation *inInfo =
88  inputVector[0]->GetInformationObject(0);
89 
90  // Any data previously owned is erased
91  this -> Clear();
92 
93  // call ExecuteData
94  vtkPolyData * input_unclean = vtkPolyData::SafeDownCast(inInfo->Get(vtkDataObject::DATA_OBJECT()));
95  vtkSmartPointer<vtkCleanPolyData> cleaner =
96  vtkSmartPointer<vtkCleanPolyData>::New();
97  cleaner -> SetInputData (input_unclean);
98  cleaner -> SetOutputPointsPrecision ( vtkAlgorithm::DesiredOutputPrecision::DOUBLE_PRECISION );
99  cleaner -> Update();
100 
101  vtkPolyData * input = cleaner -> GetOutput();
102 
103  input -> GetBounds(this -> bounds);
104 
105  this -> bounds[0] *= this -> scaleFactor;
106  this -> bounds[1] *= this -> scaleFactor;
107  this -> bounds[2] *= this -> scaleFactor;
108  this -> bounds[3] *= this -> scaleFactor;
109  this -> bounds[4] *= this -> scaleFactor;
110  this -> bounds[5] *= this -> scaleFactor;
111 
112 
113  arma::vec::fixed<3> bbox_min = {this -> bounds[0],this -> bounds[2],this -> bounds[4]};
114  arma::vec::fixed<3> bbox_max = {this -> bounds[1],this -> bounds[3],this -> bounds[5]};
115 
116 
117  vtkIdType cellId, numCells, numPts, numIds;
118 
119  numCells = input -> GetNumberOfCells();
120  numPts = input -> GetNumberOfPoints();
121  if (numCells < 1 || numPts < 1){
122  vtkErrorMacro( << "No data to measure...!");
123  return 1;
124  }
125 
126  vtkSmartPointer<vtkIdList> ptIds = vtkSmartPointer<vtkIdList>::New();
127  ptIds->Allocate(VTK_CELL_SIZE);
128 
129  // The facets are browsed to created the facet dyads
130  for (cellId=0; cellId < numCells; cellId++){
131  if ( input->GetCellType(cellId) != VTK_TRIANGLE){
132  vtkWarningMacro(<< "Input data type must be VTK_TRIANGLE not "
133  << input->GetCellType(cellId));
134  continue;
135  }
136 
137  input->GetCellPoints(cellId,ptIds);
138  numIds = ptIds->GetNumberOfIds();
139  assert(numIds == 3);
140 
141  }
142  // Generate normals
143  vtkSmartPointer<vtkPolyDataNormals> normalGenerator = vtkSmartPointer<vtkPolyDataNormals>::New();
144 
145  normalGenerator->SetInputData(input);
146  normalGenerator->ComputePointNormalsOff();
147  normalGenerator->ComputeCellNormalsOn();
148  normalGenerator->Update();
149 
150  vtkPolyData * input_with_normals = normalGenerator -> GetOutput();
151  vtkFloatArray * normals = vtkFloatArray::SafeDownCast(input_with_normals->GetCellData()->GetArray("Normals"));
152 
153  // Required by vtkPolyData::GetPointCells
154  input -> BuildLinks();
155 
156 
157  // The vertex coordinates are extracted
158  this -> vertices = new double * [input -> GetNumberOfPoints()];
159 
160  #pragma omp parallel for
161  for(int i = 0; i < input -> GetNumberOfPoints(); ++i) {
162  this -> vertices[i] = new double[3];
163  input -> GetPoint(i,this -> vertices[i]);
164  }
165 
166  // The facet are created
167  this -> facet_normals = new double * [numCells];
168  this -> facets = new int * [numCells];
169 
170 
171  #pragma omp parallel for
172  for(int i = 0; i < numCells; ++i) {
173  this -> facet_normals[i] = new double[3];
174  this -> facets[i] = new int[3];
175 
176  vtkSmartPointer<vtkIdList> ptIds = vtkSmartPointer<vtkIdList>::New();
177  ptIds -> Allocate(VTK_CELL_SIZE);
178 
179  input -> GetCellPoints(i,ptIds);
180 
181  double normal[3];
182  normals -> GetTuple(i,normal);
183 
184  this -> facet_normals[i][0] = normal[0];
185  this -> facet_normals[i][1] = normal[1];
186  this -> facet_normals[i][2] = normal[2];
187  this -> facets[i][0] = ptIds -> GetId(0);
188  this -> facets[i][1] = ptIds -> GetId(1);
189  this -> facets[i][2] = ptIds -> GetId(2);
190 
191  }
192 
193  // The edges are extracted
194  vtkSmartPointer<vtkExtractEdges> extractEdges = vtkSmartPointer<vtkExtractEdges>::New();
195  extractEdges -> SetInputData(input);
196  extractEdges -> Update();
197 
198  unsigned int edge_count = extractEdges -> GetOutput() -> GetNumberOfCells();
199  std::vector<std::array<vtkIdType,4>> edge_points_ids_facet_ids(edge_count);
200 
201  // Should get rid of this map and use a vector instead
202  // This loop cannot be parallelized since GetPointCells is
203  // not thread-safe
204  for(vtkIdType i = 0; i < edge_count; i++){
205 
206  vtkSmartPointer<vtkLine> line = vtkLine::SafeDownCast(extractEdges->GetOutput()->GetCell(i));
207  edge_points_ids_facet_ids[i][0] = line->GetPointIds()->GetId(0);
208  edge_points_ids_facet_ids[i][1] = line->GetPointIds()->GetId(1);
209 
210  // We need to find the facets forming this edge
211  vtkSmartPointer<vtkIdList> facet_ids_point_1 = vtkSmartPointer<vtkIdList>::New();
212  vtkSmartPointer<vtkIdList> facet_ids_point_2 = vtkSmartPointer<vtkIdList>::New();
213 
214  input -> GetPointCells ( edge_points_ids_facet_ids[i][0],facet_ids_point_1 );
215  input -> GetPointCells ( edge_points_ids_facet_ids[i][1],facet_ids_point_2 );
216 
217 
218 
219  // Now, we find the two facet indices showing up in both facet_ids_point_1 and facet_ids_point_2
220  facet_ids_point_1 -> IntersectWith ( facet_ids_point_2 ) ;
221 
222  if (facet_ids_point_1 -> GetNumberOfIds() != 2){
223  throw(std::runtime_error("In SBGATPolyhedronGravityModel.cpp: the intersection of the facet id lists should have exactly 2 items, not " + std::to_string(facet_ids_point_1 -> GetNumberOfIds())));
224  }
225 
226  edge_points_ids_facet_ids[i][2] = facet_ids_point_1->GetId(0);
227  edge_points_ids_facet_ids[i][3] = facet_ids_point_1->GetId(1);
228 
229  }
230 
231 
232  // The edges dyads are created
233  this -> edges = new int * [edge_points_ids_facet_ids.size()];
234  this -> edge_facets_ids = new int * [edge_points_ids_facet_ids.size()];
235 
236 
237  #pragma omp parallel for
238  for(unsigned int i = 0; i < edge_points_ids_facet_ids.size(); ++i) {
239  this -> edges[i] = new int[2];
240  this -> edge_facets_ids[i] = new int[2];
241 
242  unsigned int p0_index = edge_points_ids_facet_ids[i][0];
243  unsigned int p1_index = edge_points_ids_facet_ids[i][1];
244  unsigned int fA_index = edge_points_ids_facet_ids[i][2];
245  unsigned int fB_index = edge_points_ids_facet_ids[i][3];
246 
247  this -> edges[i][0] = p0_index;
248  this -> edges[i][1] = p1_index;
249 
250  this -> edge_facets_ids[i][0] = fA_index;
251  this -> edge_facets_ids[i][1] = fB_index;
252 
253 
254  }
255 
256  this -> N_edges = edge_count;
257  this -> N_facets = numCells;
258  this -> N_vertices = input -> GetNumberOfPoints();
259 
260  // Check that the Euler characteristic == 2
261  assert (input -> GetNumberOfPoints() - edge_count + numCells == 2);
262 
263  return 1;
264 }
