129 // Description:
130 // This method measures volume, surface area, normalized shape index, the surface area, the
131 // normalized shape index, center of mass (assuming constant density) and inertia tensor
132 // (assuming constant density) of a triangle mesh.
133 
134 int SBGATSrpYorp::RequestData(
135  vtkInformation* vtkNotUsed( request ),
136  vtkInformationVector** inputVector,
137  vtkInformationVector* vtkNotUsed( outputVector )){
138  vtkInformation *inInfo =
139  inputVector[0]->GetInformationObject(0);
140 
141 
142  if (outputFileBaseName.size() == 0){
143  throw std::runtime_error("no output file was specified");
144  }
145 
146  // call ExecuteData
147  vtkPolyData * input = vtkPolyData::SafeDownCast(
148  inInfo->Get(vtkDataObject::DATA_OBJECT()));
149 
150  vtkIdType cellId, numCells, numPts, numIds;
151 
152  numCells = input->GetNumberOfCells();
153  numPts = input->GetNumberOfPoints();
154  if (numCells < 1 || numPts < 1){
155  vtkErrorMacro( << "No data to measure...!");
156  return 1;
157  }
158 
159  vtkSmartPointer<vtkIdList> ptIds = vtkSmartPointer<vtkIdList>::New();
160  ptIds->Allocate(VTK_CELL_SIZE);
161 
162  // The facets are browsed to created the facet dyads
163  for (cellId=0; cellId < numCells; cellId++){
164  if ( input->GetCellType(cellId) != VTK_TRIANGLE){
165  vtkWarningMacro(<< "Input data type must be VTK_TRIANGLE not "
166  << input->GetCellType(cellId));
167  continue;
168  }
169 
170  input->GetCellPoints(cellId,ptIds);
171  numIds = ptIds->GetNumberOfIds();
172  assert(numIds == 3);
173 
174  }
175 
176  std::vector<std::vector<double> > vertices(numPts);
177  std::vector<std::vector<int> > facets(numCells);
178 
179 
180  // The vertex coordinates are extracted
181  #pragma omp parallel for
182  for(int i = 0; i < numPts; ++i) {
183 
184  double verts[3];
185  input -> GetPoint(i,verts);
186  std::vector<double> coords = {verts[0], verts[1], verts[2]};
187  vertices[i] = coords;
188  }
189 
190  #pragma omp parallel for
191  for(int i = 0; i < numCells; ++i) {
192 
193  vtkSmartPointer<vtkIdList> ptIds = vtkSmartPointer<vtkIdList>::New();
194  ptIds -> Allocate(VTK_CELL_SIZE);
195 
196  input -> GetCellPoints(i,ptIds);
197 
198  std::vector<int> facet = {static_cast<int>(ptIds -> GetId(0)),static_cast<int>(ptIds -> GetId(1)),static_cast<int>(ptIds -> GetId(2))};
199 
200  facets[i] = facet;
201 
202  }
203 
204 
205  YORPLib::Body targetObj(vertices,facets,this -> rho,this -> spec);
206 
207  // Determine limits for voxel grid
208  double xmax = 1.01 * targetObj.getMaxDim(1);
209  double ymax = 1.01 * targetObj.getMaxDim(2);
210  double zmax = 1.01 * targetObj.getMaxDim(3);
211 
212  // Fill out Voxel Grid
213  targetObj.setVoxelGrid(xmax, ymax, zmax, this -> numVox);
214 
215  // Compute and save SRP Fourier coefficients
216  YORPLib::SRPModel targetSRP(this -> lambdaDel,
217  this -> deltaDel,
218  this -> maxFourier,
219  &targetObj,
220  this -> howManyBounces,
221  this -> numrefine);
222 
223  targetSRP.writeSRPCoeffsFile(this -> outputFileBaseName, 2*(90.0/this -> deltaDel) + 1);
224 
225  return 1;
226 }
